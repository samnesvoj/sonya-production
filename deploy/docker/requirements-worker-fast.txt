# requirements-worker-fast.txt
# ─────────────────────────────────────────────────────────────────────────────
# Runtime dependencies for the fast worker image (sonya-worker:fast).
#
# Excluded intentionally (backend-only / dev tools):
#   fastapi, uvicorn, python-multipart  — API server only
#   psycopg2, asyncpg, alembic         — DB layer (worker uses backend API)
#   pytest, black, mypy, flake8        — dev / CI only
#   jupyter, notebook, ipykernel       — not needed on GPU worker
#
# torch + torchvision are pinned and installed BEFORE this file in the
# Dockerfile so the CUDA wheel is used; they are omitted here to prevent
# pip from pulling a CPU wheel.
# ─────────────────────────────────────────────────────────────────────────────

# ── Config / HTTP ─────────────────────────────────────────────────────────────
python-dotenv
pyyaml
requests
pydantic

# ── S3 storage ────────────────────────────────────────────────────────────────
boto3
botocore

# ── Vision / video ────────────────────────────────────────────────────────────
numpy
pillow
# headless opencv — smaller than opencv-python (no GUI libs)
opencv-python-headless
# install before scenedetect so scenedetect finds cv2 without pulling full opencv
scenedetect
mediapipe

# ── ML inference ─────────────────────────────────────────────────────────────
# ultralytics uses torch already installed via CUDA wheel — pip will not re-pull
ultralytics
# faster-whisper bundles ctranslate2 CUDA support
faster-whisper

# ── External APIs ─────────────────────────────────────────────────────────────
openai
google-genai
elevenlabs
